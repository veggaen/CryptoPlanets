# ===== CryptoPlanets Environment Variables =====
# Copy this to .env in the project root

# ===== OPTIONAL: CoinGecko API Key =====
# Get free demo key at: https://www.coingecko.com/en/api/pricing
# Default: App works WITHOUT CoinGecko (uses DefiLlama for BTC stats)
# Uncomment to enable:
# COINGECKO_API_KEY=your_demo_api_key_here

# ===== OPTIONAL: CoinMarketCap API Key =====
# Used ONLY when the selected primary provider is CoinMarketCap.
# IMPORTANT: In this app, CoinMarketCap is gated as owner-only to protect credits.
# COINMARKETCAP_API_KEY=your_coinmarketcap_key_here

# ===== OPTIONAL: Owner Unlock Key (server-side) =====
# Set this to a long random string. Then visit:
#   /api/owner?key=YOUR_KEY
# once in your browser to set an HttpOnly cookie that enables owner-only features.
# PLANETCHART_OWNER_KEY=your_long_random_owner_key

# ===== REQUIRED (for nerd mode): Etherscan API Key =====
# Get free key at: https://etherscan.io/apis
# Free Tier: 5 requests/second, 100,000 requests/day
# Only used for Ethereum token detail panels (on-demand, not in main loop)
ETHERSCAN_API_KEY=your_free_etherscan_api_key_here

# ===== Notes ===== 
# - DefiLlama: No API key needed (100% free)
# - DexScreener: No API key needed (100% free)
# - GeckoTerminal: No API key needed (100% free, disabled by default)
# - CoinGecko: Optional, for enhanced BTC global stats
# - Etherscan: Required for "nerd mode" on-chain stats (Ethereum only)
